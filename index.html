function bingo(ticket, win){
  let myWins = 0;

  ticket.forEach(ticket => {
    for (let i = 0; i < ticket[0].length; i++) {
      if (ticket[0].charCodeAt(i) === ticket[1]) {
        myWins++;

      }
    }
  });

  return myWins >= win ? 'Winner!' : 'Loser!';
}
